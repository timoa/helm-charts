{{- else if contains "LoadBalancer" .Values.frontend.service.type }}

Access Multica:
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
        Watch the status with: 'kubectl get svc --namespace {{ .Release.Namespace }} -w {{ include "multica.frontend.fullname" . }}'

  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "multica.frontend.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SERVICE_IP:{{ .Values.frontend.service.port }}
{{- else if contains "NodePort" .Values.frontend.service.type }}

Access Multica:
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "multica.frontend.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "ClusterIP" .Values.frontend.service.type }}

Access Multica:
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "multica.frontend.fullname" . }} {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }}
  URL: http://localhost:{{ .Values.frontend.service.port }}
{{- end }}
